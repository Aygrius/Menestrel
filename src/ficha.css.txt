/* O .fp3-ring é uma caixa absolute inset:0, ou seja, ocupa o
   QUADRADO INTEIRO de 560x560 do selo — não só onde há medalhão
   visível. Sem pointer-events:none aqui, o .fp3-ring--inner (que
   vem DEPOIS no DOM) fica por cima de TODA essa área, incluindo os
   pontos onde os medalhões do anel EXTERNO estão posicionados —
   isso bloqueava o clique neles mesmo com item equipado (bug real,
   não confundir com slot vazio/disabled). pointer-events:none aqui
   + auto explícito no botão (.inv-slot) resolve: o clique "atravessa"
   a caixa vazia do ring e chega no botão de fato.

   ANIMAÇÃO (pedido do usuário): os dois anéis agora orbitam
   continuamente — externo num sentido, interno no sentido
   contrário, velocidades diferentes pra não parecer um bloco só
   girando. Como o giro é aplicado aqui no .fp3-ring (pai) via
   `animation`, e cada .fp3-medal (filho) usa `transform` inline só
   pra POSICIONAR no ângulo certo (--fp3-angle), o ícone de cada
   medalhão giraria junto com o anel se nada compensasse. A
   compensação mora no .inv-slot (ver bloco abaixo): uma animação
   de rotação companion, no sentido EXATAMENTE contrário e na MESMA
   duração do anel, mantém cada ícone sempre "de pé" enquanto o
   anel gira por baixo dele — efeito de engrenagens orbitando sem
   girar o conteúdo. Respeita prefers-reduced-motion. */
#root .menestrel-ui .fp3-ring {
  position: absolute;
  inset: 0;
  pointer-events: none;
}
#root .menestrel-ui .fp3-ring--outer {
  animation: fp3-orbit-cw 90s linear infinite;
}
#root .menestrel-ui .fp3-ring--inner {
  animation: fp3-orbit-ccw 70s linear infinite;
}
@keyframes fp3-orbit-cw  { from { transform: rotate(0deg);   } to { transform: rotate(360deg);  } }
@keyframes fp3-orbit-ccw { from { transform: rotate(0deg);   } to { transform: rotate(-360deg); } }
@media (prefers-reduced-motion: reduce) {
  #root .menestrel-ui .fp3-ring--outer,
  #root .menestrel-ui .fp3-ring--inner {
    animation: none;
  }
}

/* Cada medalhão é posicionado por --fp3-angle (setado inline por
   item em JS) através de uma rotação dupla: rotate(angle) ->
   translate(raio) -> rotate(-angle). Isso desloca o medalhão pro
   ponto certo do círculo SEM rodar o conteúdo dele (ícone fica
   sempre "de pé"). O raio difere por anel (--fp3-r). Diferença de
   ~75px entre os dois raios garante respiro real entre os anéis —
   sem isso, medalhão interno e externo podiam disputar a mesma
   área de clique em certos ângulos.
   IMPORTANTE: esta regra (position:absolute + transform) é o que
   de fato tira o medalhão do fluxo normal e o coloca no círculo —
   sem ela, os 18 medalhões empilham em coluna normal (bug já visto
   numa edição anterior deste arquivo). Não remover. */
#root .menestrel-ui .fp3-medal {
  position: absolute;
  top: 50%;
  left: 50%;
  width: 0;
  height: 0;
  transform: rotate(var(--fp3-angle)) translate(var(--fp3-r)) rotate(calc(var(--fp3-angle) * -1));
}
#root .menestrel-ui .fp3-ring--inner .fp3-medal { --fp3-r: 160px; }
#root .menestrel-ui .fp3-ring--outer .fp3-medal { --fp3-r: 245px; }
@media (max-width: 599px) {
  #root .menestrel-ui .fp3-ring--inner .fp3-medal { --fp3-r: 27vw; }
  #root .menestrel-ui .fp3-ring--outer .fp3-medal { --fp3-r: 44vw; }
}

/* Contra-rotação animada: compensa exatamente o giro do .fp3-ring
   pai (mesma duração, sentido oposto ao keyframe do anel) — vem
   DEPOIS da regra de tamanho fixo acima de propósito, pois ambas
   disputam a propriedade `transform` no mesmo elemento com a MESMA
   especificidade (3 classes); na cascata, a regra que vem depois no
   arquivo vence em caso de empate. Sem essa ordem, a transform
   estática (translate(-50%,-50%)) cancelaria a animação. Os
   keyframes repetem o translate(-50%,-50%) pra não perder a
   centralização do medalhão enquanto giram. Resultado: o anel gira,
   a posição orbital do medalhão muda com ele, mas o ícone dentro do
   medalhão fica sempre "de pé", nunca de cabeça pra baixo. */
#root .menestrel-ui .fp3-ring--outer .fp3-medal .inv-slot {
  animation: fp3-counter-ccw 90s linear infinite;
}
#root .menestrel-ui .fp3-ring--inner .fp3-medal .inv-slot {
  animation: fp3-counter-cw 70s linear infinite;
}
@keyframes fp3-counter-ccw {
  from { transform: translate(-50%, -50%) rotate(0deg);   }
  to   { transform: translate(-50%, -50%) rotate(-360deg); }
}
@keyframes fp3-counter-cw {
  from { transform: translate(-50%, -50%) rotate(0deg);  }
  to   { transform: translate(-50%, -50%) rotate(360deg); }
}
@media (prefers-reduced-motion: reduce) {
  #root .menestrel-ui .fp3-ring--outer .fp3-medal .inv-slot,
  #root .menestrel-ui .fp3-ring--inner .fp3-medal .inv-slot {
    animation: none;
  }
}
